## Server Setup

```go
// cmd/rsa-server/main.go
// @title           Redis Sentinel Admin API
// @version         1.0
// @description     Monitoring and operations API for Redis Sentinel clusters on Kubernetes.
// @contact.name    Internal Infra Team
// @host            localhost:8080
// @BasePath        /api/v1
// @schemes         http

package main

import (
    "context"
    "os/signal"
    "syscall"

    "github.com/labstack/echo/v4"
    "github.com/labstack/echo/v4/middleware"
    echoSwagger "github.com/swaggo/echo-swagger"
    _ "github.com/your-org/redis-sentinel-admin/docs/swagger" // generated by swag init
)

func main() {
    e := echo.New()
    e.HideBanner = true

    // Middleware
    e.Use(middleware.Recover())
    e.Use(middleware.RequestLoggerWithConfig(...)) // see logging section
    e.Use(middleware.CORS())

    // Swagger UI — available at /swagger/index.html
    e.GET("/swagger/*", echoSwagger.WrapHandler)

    // Register all routes
    api.RegisterRoutes(e, deps)

    // Graceful shutdown
    ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
    defer stop()
    go func() {
        if err := e.Start(fmt.Sprintf(":%d", cfg.HTTPPort)); err != nil {
            e.Logger.Info("shutting down server")
        }
    }()
    <-ctx.Done()
    shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()
    e.Shutdown(shutdownCtx)
}
```

## Unified Response Format

**Every** response uses this wrapper — no exceptions:

```go
// internal/api/response.go
package api

type APIResponse struct {
    Data  interface{} `json:"data"`
    Error *APIError   `json:"error"`
}

type APIError struct {
    Code    string `json:"code"`
    Message string `json:"message"`
}

func OK(c echo.Context, data interface{}) error {
    return c.JSON(http.StatusOK, APIResponse{Data: data, Error: nil})
}

func Err(c echo.Context, status int, code, message string) error {
    return c.JSON(status, APIResponse{
        Data:  nil,
        Error: &APIError{Code: code, Message: message},
    })
}

func HandleErr(c echo.Context, err error) error {
    switch {
    case errors.Is(err, sentinel.ErrNoMaster):
        return Err(c, http.StatusServiceUnavailable, "NO_MASTER", err.Error())
    case errors.Is(err, sentinel.ErrQuorumNotMet):
        return Err(c, http.StatusServiceUnavailable, "QUORUM_NOT_MET", err.Error())
    case errors.Is(err, sentinel.ErrNodeUnreachable):
        return Err(c, http.StatusBadGateway, "NODE_UNREACHABLE", err.Error())
    case errors.Is(err, context.DeadlineExceeded):
        return Err(c, http.StatusGatewayTimeout, "TIMEOUT", "Redis operation timed out")
    default:
        return Err(c, http.StatusInternalServerError, "INTERNAL_ERROR", err.Error())
    }
}
```

## Swaggo Annotation Rules

**Every handler must have annotations.** No annotation = endpoint missing from Swagger UI.

### Standard GET handler

```go
// internal/api/handlers/topology.go

// GetTopology godoc
// @Summary      Get cluster topology
// @Description  Returns current Master, Replica, and Sentinel node states including quorum status
// @Tags         topology
// @Produce      json
// @Success      200  {object}  api.APIResponse{data=sentinel.TopologySnapshot}
// @Failure      503  {object}  api.APIResponse{error=api.APIError}  "No master or quorum not met"
// @Failure      504  {object}  api.APIResponse{error=api.APIError}  "Redis timeout"
// @Router       /topology [get]
func GetTopology(svc sentinel.Service) echo.HandlerFunc {
    return func(c echo.Context) error {
        ctx, cancel := context.WithTimeout(c.Request().Context(), 5*time.Second)
        defer cancel()

        snapshot, err := svc.GetTopology(ctx)
        if err != nil {
            return api.HandleErr(c, err)
        }
        return api.OK(c, snapshot)
    }
}
```

### GET with query parameters

```go
// GetHotkeys godoc
// @Summary      List hot keys
// @Description  Returns top N keys by access frequency using Redis LFU. Requires maxmemory-policy=allkeys-lfu or volatile-lfu.
// @Tags         keys
// @Produce      json
// @Param        top  query  int  false  "Number of top keys to return (default 20, max 100)"  default(20)
// @Success      200  {object}  api.APIResponse{data=[]keys.KeyReport}
// @Failure      400  {object}  api.APIResponse{error=api.APIError}  "LFU policy not enabled"
// @Failure      500  {object}  api.APIResponse{error=api.APIError}
// @Router       /keys/hotkeys [get]
func GetHotkeys(svc keys.Service) echo.HandlerFunc {
    return func(c echo.Context) error {
        top := 20
        if v := c.QueryParam("top"); v != "" {
            if n, err := strconv.Atoi(v); err == nil && n > 0 && n <= 100 {
                top = n
            }
        }

        ctx, cancel := context.WithTimeout(c.Request().Context(), 10*time.Second)
        defer cancel()

        result, err := svc.GetHotkeys(ctx, top)
        if err != nil {
            return api.HandleErr(c, err)
        }
        return api.OK(c, result)
    }
}
```

### POST with request body

```go
// internal/api/dto/requests.go — all request structs live here for swaggo to find

// FailoverRequest represents a failover trigger request
// @Description  Failover request payload
type FailoverRequest struct {
    DryRun  bool `json:"dry_run"  example:"false"`
    Confirm bool `json:"confirm"  example:"true"`
}

// FailoverResponse represents the result of a failover operation
// @Description  Failover result
type FailoverResponse struct {
    NewMaster   string        `json:"new_master"    example:"10.0.0.3:6379"`
    ElapsedMs   int64         `json:"elapsed_ms"    example:"342"`
    DryRun      bool          `json:"dry_run"       example:"false"`
    PreChecks   []string      `json:"pre_checks"`
}

// TriggerFailover godoc
// @Summary      Trigger graceful failover
// @Description  Executes a graceful failover: checks replica lag, selects best candidate, drains connections, promotes, then verifies. Use dry_run:true to preview without executing.
// @Tags         operations
// @Accept       json
// @Produce      json
// @Param        request  body      dto.FailoverRequest   true  "Failover options"
// @Success      200      {object}  api.APIResponse{data=dto.FailoverResponse}
// @Failure      400      {object}  api.APIResponse{error=api.APIError}  "confirm not set"
// @Failure      503      {object}  api.APIResponse{error=api.APIError}  "No eligible replica"
// @Router       /ops/failover [post]
func TriggerFailover(svc operations.Service) echo.HandlerFunc {
    return func(c echo.Context) error {
        var req dto.FailoverRequest
        if err := c.Bind(&req); err != nil {
            return api.Err(c, http.StatusBadRequest, "INVALID_REQUEST", err.Error())
        }
        if !req.DryRun && !req.Confirm {
            return api.Err(c, http.StatusBadRequest, "CONFIRM_REQUIRED",
                "set confirm:true to execute, or dry_run:true to preview")
        }

        result, err := svc.Failover(c.Request().Context(), req.DryRun)
        if err != nil {
            return api.HandleErr(c, err)
        }
        return api.OK(c, result)
    }
}
```

### SSE streaming endpoint

```go
// EventStream godoc
// @Summary      Live Sentinel event stream
// @Description  Server-Sent Events stream of real-time Sentinel events (+sdown, -sdown, +odown, +failover-*). Connect and keep the connection open to receive events as they happen.
// @Tags         events
// @Produce      text/event-stream
// @Success      200  {string}  string  "SSE stream — event: sentinel\\ndata: {json}\\n\\n"
// @Router       /events/stream [get]
func EventStream(bus *event.Bus) echo.HandlerFunc {
    return func(c echo.Context) error {
        c.Response().Header().Set(echo.HeaderContentType, "text/event-stream")
        c.Response().Header().Set("Cache-Control", "no-cache")
        c.Response().Header().Set("Connection", "keep-alive")
        c.Response().Header().Set("X-Accel-Buffering", "no") // disable nginx buffering
        c.Response().WriteHeader(http.StatusOK)

        sub := bus.Subscribe()
        defer bus.Unsubscribe(sub)

        ctx := c.Request().Context()
        for {
            select {
            case evt, ok := <-sub:
                if !ok {
                    return nil
                }
                data, _ := json.Marshal(evt)
                fmt.Fprintf(c.Response(), "event: sentinel\ndata: %s\n\n", data)
                c.Response().Flush()
            case <-ctx.Done():
                return nil
            }
        }
    }
}
```

## DTO Structs Convention

All request/response structs referenced in swaggo annotations go in `internal/api/dto/`:

```
internal/api/dto/
├── requests.go    # All request body structs
└── responses.go   # All response-specific structs (not already defined in domain packages)
```

Domain structs (e.g. `sentinel.TopologySnapshot`, `keys.KeyReport`) can be referenced directly in annotations — no need to duplicate.

## Custom Error Handler (Echo global)

```go
e.HTTPErrorHandler = func(err error, c echo.Context) {
    // Handle Echo's own HTTPError (e.g. 404 from router)
    var he *echo.HTTPError
    if errors.As(err, &he) {
        api.Err(c, he.Code, "HTTP_ERROR", fmt.Sprintf("%v", he.Message))
        return
    }
    // Fallback
    api.Err(c, http.StatusInternalServerError, "INTERNAL_ERROR", err.Error())
}
```
